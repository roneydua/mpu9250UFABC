216 void EKF::updateStage() {
217  // Ganha de Kalman
218  K.topLeftCorner(n, p) = P.topLeftCorner(n, n) *
219  H.topLeftCorner(p, n).transpose() *
220  (H.topLeftCorner(p, n) * P.topLeftCorner(n, n) *
221  (H.topLeftCorner(p, n)).transpose() +
222  R.topLeftCorner(p, p))
223  .inverse();
224  // Atualiza a matriz de covariância.
225  P.topLeftCorner(n, n) = (Eigen::MatrixXf::Identity(n, n) -
226  K.topLeftCorner(n, p) * H.topLeftCorner(p, n)) *
227  P.topLeftCorner(n, n);
228  // Computa o erro de estimação #deltaChi
229  deltaChi.segment(1, n) = K.topLeftCorner(n, p) * deltaYObs.head(p);
230  updateStates();
231 }
