  22 Eigen::Matrix3f skew(const Eigen::Vector3f &v) {
  23 
  24   Eigen::Matrix3f M = Eigen::Matrix3f::Zero();
  25   M << 0, -v(2), v(1), v(2), 0, -v(0), -v(1), v(0), 0;
  26   return M;

  29  * @brief Calcula a Matriz _Right-Quaternion_ a partir do quaternion q
  30  * @param  q quaternion de atitude
  31  * @return mS_r Matrix _Right-Quaternion_
  32  * @brief A matrix _Right-Quaternion_ torna poss√≠vel realizar o produto de

  35 Eigen::Matrix4f S_r(const Eigen::Vector4f &q) {
  36 
  37   Eigen::Matrix4f mS_r = Eigen::Matrix4f::Identity() * q(0);
  38   mS_r.block<1, 3>(0, 1) = -q.segment<3>(1);
  39   mS_r.block<3, 1>(1, 0) = q.segment<3>(1);
  40   mS_r.block<3, 3>(1, 1) -= skew(q.segment<3>(1));
  41   return mS_r;
  42 }

  48 void computeMcdFromQuaternion(const Eigen::Vector4f &q, Eigen::Matrix3f &mcd) {
  49   mcd = Eigen::Matrix3f::Identity() * (2.0f * q(0) * q(0) - 1.0f) +
  50         2.0f * (q.tail(3) * q.tail(3).transpose()) +
  51         2.0f * q(0) * skew(q.tail(3));
  52 }

  79  * @brief Produto de quaternions
  80  * @param  p Quaternion
  81  * @param  q Quaternion
  82  * @return   Quaternion product p&otimes;q
  83  */
  84 Eigen::Vector4f multiplyQuaternions(const Eigen::Vector4f &p,
  85                                     const Eigen::Vector4f &q) {
  86   Eigen::Vector4f pq;
  87   pq(0) = p(0) * q(0) - p(1) * q(1) - p(2) * q(2) - p(3) * q(3);
  88   pq(1) = p(0) * q(1) + p(1) * q(0) + p(2) * q(3) - p(3) * q(2);
  89   pq(2) = p(0) * q(2) - p(1) * q(3) + p(2) * q(0) + p(3) * q(1);
  90   pq(3) = p(0) * q(3) + p(1) * q(2) - p(2) * q(1) + p(3) * q(0);
  91   return pq;
  92 }
  93 /**
  94  * @brief Produto de quaternions
  95  * @param  p Quaternion
  96  * @param  q Vetor tridimensional
  97  * @return Quaternion product p&otimes;q
  98  */
  99 Eigen::Vector4f multiplyQuaternions(const Eigen::Vector4f &p,
 100                                     const Eigen::Vector3f &q) {
 101   Eigen::Vector4f pq;
 102   pq(0) = -p(1) * q(0) - p(2) * q(1) - p(3) * q(2);
 103   pq(1) = p(0) * q(0) + p(2) * q(2) - p(3) * q(1);
 104   pq(2) = p(0) * q(1) - p(1) * q(2) + p(3) * q(0);
 105   pq(3) = p(0) * q(2) + p(1) * q(1) - p(2) * q(0);
 106   return pq;
 107 }

 114 void quaternion2Euler(Eigen::Vector3f &_euler, const Eigen::Vector4f &_q) {
 115   // calculo de fi
 116   _euler(0) = RAD_TO_DEG * atan2f((_q(0) * _q(1) + _q(2) * _q(3)),
 117                                   0.5f - (_q(1) * _q(1) + _q(2) * _q(2)));
 118   // calculo de theta
 119   _euler(1) = RAD_TO_DEG * asinf(2.0f * (_q(0) * _q(1) - _q(0) * _q(3)));
 120   // calculo de psi
 121   _euler(2) = RAD_TO_DEG * atan2f((_q(0) * _q(3) + _q(1) * _q(2)),
 122                                   0.5f - (_q(2) * _q(2) + _q(3) * _q(3)));
 123 }
