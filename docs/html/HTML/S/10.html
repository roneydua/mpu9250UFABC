  10 #ifndef METODOSCALIBRACAO_H_
  11 #define METODOSCALIBRACAO_H_

  19 using namespace Eigen;
  20 // comente ou descomente a linha abaixo dependendo da plataforma utilizada
  21 class METODOSCALIBRACAO {
  22 protected:
  23 public:
  24   /**
  25   * Metodo geometrico para calibracao de magnetometros
  26   * @param #data <Matrix com N linha e 3 colunas contendo os
  27   dados de amostra para a calibracao> [Matrix de scala] sF Matrix 3x3 com
  28   diagonal contendo os fatores de escalas
  29   @param bias Vector 3x1 with the bias - ellipsoid center
  30   @param moduloCampo values in uT of earth field
  31   * @return [Matriz com fatores de escala na diagonal] sF
  32                                  [vetor com o centro da elipsoide] <bias>
  33 
  34   * The method fitting an ellipsoid with least squares method. The compensated
  35   measures  M_c are obtained with the mensures M proceeding as follows: M_c = sF
  36   * (M - bias) The method needs a full rotations for a good performance
  37   */
  38   int calibracaoGeometrica(MatrixXf &data, Matrix3f &sF, Vector3f &bias,
  39                            float moduloCampo) {
  40 
  41     // data eh uma matrix com N x 3, sendo N o numero de medidas coletadas
  42     if (data.rows() == 3) {
  43       data.transposeInPlace();
  44     }
  45     sF = Matrix3f::Identity();
  46     bias = Vector3f::Zero();
  47     int N = data.rows();
  48     MatrixXf H = MatrixXf::Ones(N, 6);
  49     VectorXf w(N), X(6), tmp(3);
  50     H.col(0) = data.col(0);
  51     H.col(1) = data.col(1);
  52     H.col(2) = data.col(2);
  53     H.col(3) = -data.col(1).array().pow(2).matrix();
  54     H.col(4) = -data.col(2).array().pow(2).matrix();
  55     w = data.col(0).array().pow(2).matrix();
  56     data.resize(0, 0); // libera memoria
  57     // X = (H.transpose() * H).inverse() * H.transpose() * w;
  58     X = H.fullPivHouseholderQr().solve(w);
  59     H.resize(0, 0); // libera memoria
  60     bias(0) = X(0) / 2.0;
  61     bias(1) = X(1) / (2.0 * X(3));
  62     bias(2) = X(2) / (2.0 * X(4));
  63     tmp(0) = (X(5) + bias(0) * bias(0) + X(3) * bias(1) * bias(1) +
  64               X(4) * bias(2) * bias(2)) /
  65              (moduloCampo * moduloCampo);
  66     tmp(1) = tmp(0) / X(3);
  67     tmp(2) = tmp(0) / X(4);
  68     sF(0, 0) = 1.0 / sqrt(tmp(0));
  69     sF(1, 1) = 1.0 / sqrt(tmp(1));
  70     sF(2, 2) = 1.0 / sqrt(tmp(2));
  71 // checa se ha numero INFINITOS ou INVALIDOS retornoando -2
  72 // quando encontra numeros NaN e -1 para numeros insfinitos
  73 #ifndef __GNUC__
  74     for (int i = 0; i < 3; i++) {
  75       for (size_t j = 0; j < 3; j++) {
  76         if (isnan(sF(i, j)) || isinf(sF(i, j))) {
  77           return -2;
  78         }
  79       }
  80       if (isnan(bias(i)) || isinf(bias(i))) {
  81         return -1;
  82       }
  83     }
  84 #endif
  85     return 1;
  86   }
  87   /**
  88    * Calibração do acelerômetro. Os valores compensador sao obtidos:
  89    *  **Yc = sF * (Ym - bias)**
  90    * Ym: medidas diretas do sensor
  91    * Yc: meddias compensadas
  92    * @param sF (MatriX3f) Referência da matrix que contem corrige os
  93    * fatores de escala e dos desalinhamentos.
  94    * @param bias (int16_t array)
  95    * @param data Medidas para calibração de dimensão n&times;3
  96    * @return 1 para positivo e 0 para a falha.
  97    */
  98   int calibracaoAcelerometro(Matrix3f &sF, int16_t (&bias)[3], MatrixXf data) {
  99     // data eh uma matrix com N x 3, sendo N o numero de medidas coletadas
 100     if (data.rows() == 3) {
 101       data.transposeInPlace();
 102     }
 103     int indice;
 104     int N = data.rows();
 105     MatrixXf Y = MatrixXf::Zero(N, 3);
 106     MatrixXf w = MatrixXf::Ones(N, 4);
 107     MatrixXf X = MatrixXf::Zero(4, 3);
 108     Vector3f _bias;
 109     for (int var = 0; var < N; var++) {
 110       data.row(var).maxCoeff(&indice);
 111       if (data(var, indice) > 1000.0f) {
 112         Y(var, indice) = 9.80665f;
 113         w.row(var) << data.row(var), 1.0;
 114       } else {
 115         data.row(var).minCoeff(&indice);
 116         Y(var, indice) = -9.80665f;
 117         w.row(var) << data.row(var), 1.0;
 118       }
 119     }
 120     X = w.colPivHouseholderQr().solve(Y);
 121     X.transposeInPlace();
 122 #ifndef __GNUC__
 123     for (int i = 0; i < X.rows(); ++i) {
 124       for (int j = 0; j < X.cols(); ++j) {
 125         if (isnan(X(i, j)) || isinf(X(i, j))) {
 126           return 0;
 127         }
 128       }
 129     }
 130 #endif
 131     sF = X.block(0, 0, 3, 3);
 132     _bias = -sF.inverse() * X.col(3);
 133     // bias
 134     // bias = -X.col(3);
 135     bias[0] = (int16_t)_bias[0];
 136     bias[1] = (int16_t)_bias[1];
 137     bias[2] = (int16_t)_bias[2];
 138     return 1;
 139   }
 140 };
