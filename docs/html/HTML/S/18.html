  49 /**
  50  * Loop do Filtro de Kalaman
  51  */
  52 void EKF::loopEKF() {
  53   // Faz a leitura dos sensores da MARG atualizando os vetores #accel, #gyro e
  54   // #mag.
  55   imu.readSensor();
  56   // tempo decorrido entre medidas a medida anterior e a atual.
  57   dt = timeClass.computeElapsedTime();
  58   // Esta de predição.
  59   temporalPrediction();
  60 
  61   updateH();
  62   // Ganha de Kalman
  63   K.topLeftCorner(n, p) = P.topLeftCorner(n, n) *
  64                           H.topLeftCorner(p, n).transpose() *
  65                           (H.topLeftCorner(p, n) * P.topLeftCorner(n, n) *
  66                                (H.topLeftCorner(p, n)).transpose() +
  67                            R.topLeftCorner(p, p))
  68                               .inverse();
  69   // Atualiza a matriz de covariância.
  70   P.topLeftCorner(n, n) = (Eigen::MatrixXf::Identity(n, n) -
  71                            K.topLeftCorner(n, p) * H.topLeftCorner(p, n)) *
  72                           P.topLeftCorner(n, n);
  73   // Serial.println("deltaYObs.head(p)================");
  74   // printEigen(deltaYObs.head(p));
  75   // deltaChi.head(p) = K.topLeftCorner(p, p) * deltaYObs.head(p);
  76   // Corrige os estados com deltaChi.
  77   statesUpdate();
  78 }

 104 void EKF::temporalPrediction() {
 105   /*! @attention dcm é uma referência. */
 106   // Atualiza matriz de rotação com a ultima atitude q_{k-1}
 107   Serial.println("================== entrou na predicao ================== ");
 108   computeMcdFromQuaternion(q, mcd);
 109   // updateOfMeasurements();
 110   Serial.printf("Tempo \t%f\n", dt);
 111   localGravity = -mcd.transpose() * deltar + accel;
 112   magOrthogonalGrav = computeVectorProjection(mag, localGravity);
 113   // atualiza as matrizes F, fi e G  com os estados atuais.
 114   updatefi();
 115   // Integracao dos estados
 116   integrationOfStates();
 117   // atualiza a matriz de covariâncias
 118   P.topLeftCorner(p, p) = fi.topLeftCorner(p, p) * P.topLeftCorner(p, p) *
 119                               fi.topLeftCorner(p, p).transpose() +
 120                           dt * G.topLeftCorner(p, p) * Q.topLeftCorner(p, p) *
 121                               G.topLeftCorner(p, p).transpose();
 122 }

 167 void EKF::updateF_G() {
 168   /*! \HACK @hack Apenas parte da matrix é atualizada*/
 169   F.topLeftCorner(3, 3) = -skew(gyro);
 170   /*! @TODO A matriz GQG* tem apenas uma parte modificada em cada
 171    * iteração e é usada apenas na propagação da covariância. Pode-se otimizar
 172    * gerando apenas uma matriz e reduzindo o numero de multiplicações.*/
 173   if (p > 6) {
 174     /*! @note A única parte atulizada da matriz é o bloco 3x3  quando há
 175      * medidas do gps apenas.*/
 176 
 177     G.bottomRightCorner(3, 3) = -mcd;
 178     F.bottomLeftCorner(3, 3) = -2.0 * skew(mcd * accel);
 179   }
 180 }

 186 void EKF::updatefi() {

 192 /**
 193  * Atualiza matrix de sessitividade #H.
 194  * @param g Vetor gravitacional predito.
 195  * @param m Campo magnético predito.
 196  */
 197 void EKF::updateH() {
 198   /*! @warning Multiplicado apenas a respectiva coluna da matrix #dcm
 199     @todo Esta parte pode ser otimizada.
 200   */
 201   Eigen::Vector3f predictedGravityField =
 202       navegationGravity(2) * (mcd.transpose()).col(2);
 203 
 204   float xy = magOrthogonalGrav.head(2).norm();
 205 
 206   Eigen::Vector3f predictedMagneticField =
 207       xy * (mcd.transpose()).col(0) +
 208       magOrthogonalGrav(2) * (mcd.transpose()).col(2);
 209 
 210   deltaYObs.head(3) = localGravity - predictedGravityField;
 211   deltaYObs.segment(3, 3) = mag - predictedMagneticField;
 212   /*! @note
 213   Atualiza as 6 primeiras linas das três primeiras colunas.
 214   */
 215   H.topLeftCorner(3, 3) = 2.0f * skew(predictedGravityField);
 216   H.block<3, 3>(3, 0) = 2.0f * skew(predictedMagneticField);
 217 }
 218 /**
 219  * Integra os estado #chi;.
 220  */
 221 void EKF::integrationOfStates() {
 222   // Quatérnio com Euler de primeira ordem
 223   q += 0.5 * dt * multiplyQuaternions(q, gyro);
 224   // atualiza matriz de rotação com o quaternio predito
 225   computeMcdFromQuaternion(q, mcd);
 226   // Bias do giroscópio.
 227   /*!  @warning Modifica-se o bias da classe imu.*/
 228   imu._biasGyro += deltaChi.segment(3, 3);
 229   // Acelacao linear
 230   deltar = -navegationGravity + mcd * accel;
 231   // Velocidade linear
 232   r += dt * deltar;
 233 }
