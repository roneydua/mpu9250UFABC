 246 // The last two bytes of the message is a checksum value, used to confirm that
 247 // the received payload is valid. The procedure used to calculate this is given
 248 // as pseudo-code in the uBlox manual.
 249 void calcChecksum(unsigned char *CK, int msgSize) {
 250   memset(CK, 0, 2);
 251   for (int i = 0; i < msgSize; i++) {
 252     CK[0] += ((unsigned char *)(&ubxMessage))[i];
 253     CK[1] += CK[0];
 254   }
 255 }

 257 // Compares the first two bytes of the ubxMessage struct with a specific message
 258 // header. Returns true if the two bytes match.
 259 boolean compareMsgHeader(const unsigned char *msgHeader) {
 260   unsigned char *ptr = (unsigned char *)(&ubxMessage);
 261   return ptr[0] == msgHeader[0] && ptr[1] == msgHeader[1];
 262 }

 264 int processGPS() {
 265   static int fpos = 0;
 266   static unsigned char checksum[2];
 267 
 268   static byte currentMsgType = MT_NONE;
 269   static int payloadSize = sizeof(UBXMessage);
 270   // trocado while por if para naum travar o processo.
 271   while (GPSSERIAL.available()) {
 272 
 273     byte c = GPSSERIAL.read();
 274     // Serial.write(c);
 275 
 276     if (fpos < 2) {
 277       /*
 278       Procura se os primeiro dois bytes sao os cabecalhos(0xB5,0x62)
 279        */
 280       if (c == UBX_HEADER[fpos])
 281         fpos++;
 282       else
 283         fpos = 0;
 284     } else {
 285       // If we come here then fpos >= 2, which means we have found a match with
 286       // the UBX_HEADER and we are now reading in the bytes that make up the
 287       // payload.
 288 
 289       // Place the incoming byte into the ubxMessage struct. The position is
 290       // fpos-2 because the struct does not include the initial two-byte header
 291       // (UBX_HEADER).
 292       if ((fpos - 2) < payloadSize)
 293         ((unsigned char *)(&ubxMessage))[fpos - 2] = c;
 294 
 295       fpos++;
 296 
 297       if (fpos == 4) {
 298         // We have just received the second byte of the message type header,
 299         // so now we can check to see what kind of message it is.
 300 #ifdef NAV_POSLLH
 301         if (compareMsgHeader(NAV_POSLLH_HEADER)) {
 302           currentMsgType = MT_NAV_POSLLH;
 303           payloadSize = sizeof(NAV_POSLLH_STRUCT);
 304         } else
 305 #endif
 306 #ifdef NAV_VELNED
 307             if (compareMsgHeader(NAV_VELNED_HEADER)) {
 308           currentMsgType = MT_NAV_VELNED;
 309           payloadSize = sizeof(NAV_VELNED_STRUCT);
 310         } else
 311 #endif
 312         {
 313           // unknown message type, bail
 314           fpos = 0;
 315           continue;
 316         }
 317       }
 318 
 319       if (fpos == (payloadSize + 2)) {
 320         // All payload bytes have now been received, so we can calculate the
 321         // expected checksum value to compare with the next two incoming bytes.
 322         calcChecksum(checksum, payloadSize);
 323       } else if (fpos == (payloadSize + 3)) {
 324         // First byte after the payload, ie. first byte of the checksum.
 325         // Does it match the first byte of the checksum we calculated?
 326         if (c != checksum[0]) {
 327           // Checksum doesn't match, reset to beginning state and try again.
 328           fpos = 0;
 329         }
 330       } else if (fpos == (payloadSize + 4)) {
 331         // Second byte after the payload, ie. second byte of the checksum.
 332         // Does it match the second byte of the checksum we calculated?
 333         fpos = 0; // We will reset the state regardless of whether the checksum
 334                   // matches.
 335         if (c == checksum[1]) {
 336           // Checksum matches, we have a valid message.
 337           return currentMsgType;
 338         }
 339       } else if (fpos > (payloadSize + 4)) {
 340         // We have now read more bytes than both the expected payload and
 341         // checksum together, so something went wrong. Reset to beginning state
 342         // and try again.
 343         fpos = 0;
 344       }
 345     }
 346   }
 347   return MT_NONE;
 348 }
 349 void gps_imprime() {
 350   while (GPSSERIAL.available()) {
 351     // Serial.write(GPSSERIAL.read());
 352     int msgType = processGPS();
 353 #ifdef NAV_VELNED
 354     if (msgType == MT_NAV_VELNED) {
 355       Serial.printf("%f\t\n", float(ubxMessage.navVelned.velD) / 100.0f);
 356     }
 357 #endif
 358 #ifdef NAV_POSLLH
 359     if (msgType == MT_NAV_POSLLH) {
 360       Serial.printf("%ld \t%ld \t\n", ubxMessage.navPosllh.lat,
 361                     ubxMessage.navPosllh.lon);
 362     }
 363 #endif
 364   }
 365 }
