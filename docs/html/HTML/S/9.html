  20 int IMU::begin(Vector3f &acel, Vector3f &gyro, Vector3f &mag) {
  21   _acel = &acel;
  22   _gyro = &gyro;
  23   _mag = &mag;
  24   // para analise da calibracao
  25   // inicializa a biblioteca Wire para comunicacao i2c
  26   _i2c->begin(21, 22);
  27 
  28   // seta a frequencia de comunicacao I2c
  29   _i2c->setClock(_i2cRate);
  30   // seleciona o "clock source" do giroscopio
  31 
  32   writeRegister(PWR_MGMNT_1, CLOCK_SEL_PLL);
  33   // habilita o modo master i2c
  34   writeRegister(USER_CTRL, I2C_MST_EN);
  35   // configura a frequencia do I2c
  36   writeRegister(I2C_MST_CTRL, I2C_MST_CLK);
  37   // desliga o magnetometro
  38   writeAK8963Register(AK8963_CNTL1, AK8963_PWR_DOWN);
  39   // reseta a MPU9250
  40   writeRegister(PWR_MGMNT_1, PWR_RESET);
  41   delay(1);
  42   // reset o magnetometro
  43   writeAK8963Register(AK8963_CNTL2, AK8963_RESET);
  44   // seleciona o "clock source" do giroscopio
  45   writeRegister(PWR_MGMNT_1, CLOCK_SEL_PLL);
  46   // habilita o acelerometro o giroscopio
  47   writeRegister(PWR_MGMNT_2, SEN_ENABLE);
  48   // selecionas as escalas de 16G e 2000DPS para acel e Giro como padrao
  49   setAccelRange(ACCEL_RANGE_16G);
  50   setGyroRange(GYRO_RANGE_2000DPS);
  51   // atribui define a largura de banda como padrao
  52   _bandwidth = DLPF_BANDWIDTH_92HZ;
  53   setDlpfBandwidth(_bandwidth);
  54   writeRegister(SMPDIV, 0x00);
  55   _srd = 0;
  56   // enable I2C master mode
  57   if (writeRegister(USER_CTRL, I2C_MST_EN) < 0) {
  58     return -12;
  59   }
  60   // set the I2C bus speed to 400 kHz
  61   if (writeRegister(I2C_MST_CTRL, I2C_MST_CLK) < 0) {
  62     return -13;
  63   }
  64   // check AK8963 WHO AM I register, expected value is 0x48 (decimal 72)
  65   if (whoAmIAK8963() != 72) {
  66     return -14;
  67   }
  68   /* get the magnetometer calibration */
  69   // set AK8963 to Power Down
  70   if (writeAK8963Register(AK8963_CNTL1, AK8963_PWR_DOWN) < 0) {
  71     return -15;
  72   }
  73   delay(100); // long wait between AK8963 mode changes
  74   // set AK8963 to FUSE ROM access
  75   if (writeAK8963Register(AK8963_CNTL1, AK8963_FUSE_ROM) < 0) {
  76     return -16;
  77   }
  78   delay(100);
  79   // para detalhes ver manual AK8963 pag. 32  e pg. 29
  80   readAK8963Registers(AK8963_ASA, 3, _buffer);
  81   _magScaleX = ((((float)_buffer[0]) - 128.0) / (256.0) + 1.0) * 4912.0 /
  82                32760.0; // micro Tesla
  83   _magScaleY = ((((float)_buffer[1]) - 128.0) / (256.0) + 1.0) * 4912.0 /
  84                32760.0; // micro Tesla
  85   _magScaleZ = ((((float)_buffer[2]) - 128.0) / (256.0) + 1.0) * 4912.0 /
  86                32760.0; // micro Tesla
  87   // set AK8963 desligao magnetometro
  88   if (writeAK8963Register(AK8963_CNTL1, AK8963_PWR_DOWN) < 0) {
  89     return -17;
  90   }
  91   delay(100); // delay para aplicacao das configuracoes
  92   // inicializa o magnetometro com resolucao de 16bit e 100Hz
  93   if (writeAK8963Register(AK8963_CNTL1, AK8963_CNT_MEAS2) < 0) {
  94     return -18;
  95   }
  96   delay(100); // delay para aplicacao das configuracoes
  97   // select clock source to gyro
  98   if (writeRegister(PWR_MGMNT_1, CLOCK_SEL_PLL) < 0) {
  99     return -19;
 100   }
 101   // instruct the MPU9250 to get 7 bytes of data from the AK8963 at the sample
 102   // rate
 103   readAK8963Registers(AK8963_HXL, 7, _buffer);
 104 
 105   return 1;
 106 }

 112 int IMU::setAccelRange(AccelRange range) {
 113   switch (range) {
 114   case ACCEL_RANGE_2G: {
 115     // setting the accel range to 2G
 116     if (writeRegister(ACCEL_CONFIG, ACCEL_FS_SEL_2G) < 0) {
 117       return -1;
 118     }
 119     _accelScale = (gG * 2.0) / 32768.0; // setting the accel scale to 2G
 120     break;
 121   }
 122   case ACCEL_RANGE_4G: {
 123     // setting the accel range to 4G
 124     if (writeRegister(ACCEL_CONFIG, ACCEL_FS_SEL_4G) < 0) {
 125       return -1;
 126     }
 127     _accelScale = (gG * 4.0f) / 32768.0f; // setting the accel scale to 4G
 128     break;
 129   }
 130   case ACCEL_RANGE_8G: {
 131     // setting the accel range to 8G
 132     if (writeRegister(ACCEL_CONFIG, ACCEL_FS_SEL_8G) < 0) {
 133       return -1;
 134     }
 135     _accelScale = (gG * 8.0f) / 32768.0f; // setting the accel scale to 8G
 136     break;
 137   }
 138   case ACCEL_RANGE_16G: {
 139     // setting the accel range to 16G
 140     if (writeRegister(ACCEL_CONFIG, ACCEL_FS_SEL_16G) < 0) {
 141       return -1;
 142     }
 143     _accelScale = (gG * 16.0f) / 32768.0f; // setting the accel scale to 16G
 144     break;
 145   }
 146   }
 147   _accelRange = range;
 148   return 1;

 277 int IMU::readSensor() {
 278   //_useSPIHS = true; // use the high speed SPI for data readout
 279   // grab the data from the MPU9250
 280   if (readRegisters(ACCEL_OUT, 21, _buffer) < 0) {
 281     return -1;
 282   }
 283   // combine into 16 bit values
 284   _axcounts = (((int16_t)_buffer[0]) << 8) | _buffer[1];
 285   _aycounts = (((int16_t)_buffer[2]) << 8) | _buffer[3];
 286   _azcounts = (((int16_t)_buffer[4]) << 8) | _buffer[5];
 287 
 288   // _tcounts = (((int16_t)_buffer[6]) << 8) | _buffer[7];
 289 
 290   _gxcounts = (((int16_t)_buffer[8]) << 8) | _buffer[9];
 291   _gycounts = (((int16_t)_buffer[10]) << 8) | _buffer[11];
 292   _gzcounts = (((int16_t)_buffer[12]) << 8) | _buffer[13];
 293 
 294   _hxcounts = (((int16_t)_buffer[15]) << 8) | _buffer[14];
 295   _hycounts = (((int16_t)_buffer[17]) << 8) | _buffer[16];
 296   _hzcounts = (((int16_t)_buffer[19]) << 8) | _buffer[18];
 297   /* corrige a direcao do eixo z  para -z e comuta o x com o y
 298 esse procedimento eh fundamental para que os eixos do acelerometro e gyro seja
 299 compartilhados como o magnetometro */
 300   // *_acel << (float)(_aycounts - _biasAccel[0]),
 301   //     (float)(_axcounts - _biasAccel[1]), (float)(_azcounts - _biasAccel[2]);
 302   *_acel << (float)(_aycounts), (float)(_axcounts), (float)(_azcounts);
 303   *_acel *= _accelScale;
 304   *_acel -= _biasAccel;
 305   // *_acel = sFAccel * (*_acel);
 306   // *_acel = sFAccel * (*_acel) + biasAccel;
 307   *_gyro << -(float)_gycounts, -(float)_gxcounts, -(float)_gzcounts;
 308   *_gyro *= _gyroScale;
 309   *_gyro -= _biasGyro;
 310   // *_mag << -(float)_hxcounts*_magScaleX, -(float)_hycounts*_magScaleX,
 311   // (float)_hzcounts*_magScaleX;
 312   // *_mag = _sM * (*_mag - _biasMag);
 313   // para analise da calibracao
 314   magTemp << -(float)_hxcounts * _magScaleX, -(float)_hycounts * _magScaleX,
 315       (float)_hzcounts * _magScaleX;
 316   *_mag = _sM * (magTemp - _biasMag);
 317   /* @annotation retirado do manual de registros da mpu9250 pag.33*/
 318   // TEMP_degC = ((TEMP_OUT – RoomTemp_Offset)/Temp_Sensitivity)+ 21degC
 319   // RoomTemp_Offset = 21; Temp_Sensitivity = 333.87
 320   // _t = (float)_tcounts / 333.87f - 20.937101267f;
 321   return 1;
 322 }

 341   writeRegister(I2C_SLV0_ADDR, AK8963_I2C_ADDR | I2C_READ_FLAG);
 342   writeRegister(I2C_SLV0_REG, subAddress);
 343   writeRegister(I2C_SLV0_CTRL, I2C_SLV0_EN | count);

 348 int IMU::writeAK8963Register(uint8_t subAddress, uint8_t data) {
 349   // set slave 0 to the AK8963 and set for write
 350   if (writeRegister(I2C_SLV0_ADDR, AK8963_I2C_ADDR) < 0) {
 351     return -1;
 352   }
 353   // set the register to the desired AK8963 sub address
 354   if (writeRegister(I2C_SLV0_REG, subAddress) < 0) {
 355     return -2;
 356   }
 357   // store the data for write
 358   if (writeRegister(I2C_SLV0_DO, data) < 0) {
 359     return -3;
 360   }
 361   // enable I2C and send 1 byte
 362   if (writeRegister(I2C_SLV0_CTRL, I2C_SLV0_EN | (uint8_t)1) < 0) {
 363     return -4;
 364   }
 365   // read the register and confirm
 366   if (readAK8963Registers(subAddress, 1, _buffer) < 0) {
 367     return -5;
 368   }
 369   if (_buffer[0] == data) {
 370     return 1;
 371   } else {
 372     return -6;
 373   }
 374 }
 375 /**
 376  * Escreve nos registros da Marg
 377  * @param  subAddress
 378  * @param  data
 379  * @return
 380  * @detail Esada para ler os dados e configurar a MARG.
 381  */
 382 int IMU::writeRegister(uint8_t subAddress, uint8_t data) {
 383   _i2c->beginTransmission(_address); // open the device
 384   _i2c->write(subAddress);           // write the register address
 385   _i2c->write(data);                 // write the data
 386   _i2c->endTransmission();
 387   delay(10);
 388   /* read back the register */
 389   readRegisters(subAddress, 1, _buffer);
 390   /* check the read back register against the written register */
 391   if (_buffer[0] == data) {
 392     return 1;
 393   } else {
 394     return -1;
 395   }
 396 }
 397 /**
 398  * Leitura dos registros da MARG
 399  * @param  subAddress
 400  * @param  count
 401  * @param  dest
 402  * @return            positivo se ok, negativo para o contrário.
 403  */
 404 int IMU::readRegisters(uint8_t subAddress, uint8_t count, uint8_t *dest) {
 405   _i2c->beginTransmission(_address);
 406   _i2c->write(subAddress);
 407   _i2c->endTransmission(false);
 408   _numBytes = _i2c->requestFrom(_address, count);
 409   if (_numBytes == count) {
 410     for (uint8_t i = 0; i < count; i++) {
 411       dest[i] = _i2c->read();
 412     }
 413     return 1;
 414   } else {
 415     return -1;
 416   }
 417 }
