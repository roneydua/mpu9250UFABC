278  Procura se os primeiro dois bytes sao os cabecalhos(0xB5,0x62)

280  if (c == UBX_HEADER[fpos])
281  fpos++;

284  } else {
285  // If we come here then fpos >= 2, which means we have found a match with

288

290  // fpos-2 because the struct does not include the initial two-byte header
291  // (UBX_HEADER).
292  if ((fpos - 2) < payloadSize)
293  ((unsigned char *)(&ubxMessage))[fpos - 2] = c;
294 
295  fpos++;
296 
297  if (fpos == 4) {
298  // We have just received the second byte of the message type header,
299  // so now we can check to see what kind of message it is.
300 #ifdef NAV_POSLLH

304  } else
305 #endif

308  currentMsgType = MT_NAV_VELNED;
309  payloadSize = sizeof(NAV_VELNED_STRUCT);
310  } else

312  {

314  fpos = 0;
315  continue;
316  }
317  }

322  calcChecksum(checksum, payloadSize);
323  } else if (fpos == (payloadSize + 3)) {
